# Тестування GET запиту (спочатку має повернути 404)
curl -X GET http://localhost:3000/200

# Тестування PUT запиту
curl -X PUT http://localhost:3000/200 -H "Content-Type: image/jpeg" --data-binary "@test-image.jpg"

# Тестування GET запиту після PUT (має повернути картинку)
curl -X GET http://localhost:3000/200 --output downloaded-200.jpg

# Тестування DELETE запиту
curl -X DELETE http://localhost:3000/200

# Тестування недозволеного методу
curl -X POST http://localhost:3000/200

# Тестування невірного статус коду
curl -X GET http://localhost:3000/abc

# Тестування автоматичного отримання з http.cat
curl -X GET http://localhost:3000/404 --output cat-404.jpg
curl -X GET http://localhost:3000/500 --output cat-500.jpg

# Перевірка що картинки зберігаються в кеші (другий запит має бути швидшим)
curl -X GET http://localhost:3000/200 --output cat-200.jpg
